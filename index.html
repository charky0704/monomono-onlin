モンスターが現れた！
"; // スライムの攻撃力を少し上げ、守備力を設定 let enemy = { name: "スライム", hp: 60, max: 60, atk: 15, matk: 5, def: 5, mdef: 5, spd: 8 }; document.getElementById('e-hp-bar').style.width = "100%"; // バトルのメインループ（1.5秒ごとに1ターン進行） const battleInterval = setInterval(() => { // AがBを攻撃する内部関数 const executeTurn = (attacker, target, isPlayerAttacking) => { if (curHP <= 0 || enemy.hp <= 0) return true; // どちらかが死んでたら中断 if (isPlayerAttacking) { if (pAtk(enemy, log)) { finish(battleInterval, true); return true; } } else { if (eAtk(enemy, log)) { finish(battleInterval, false); return true; } } return false; }; // 速度比較で順番を決める if (myStats.spd >= enemy.spd) { // プレイヤー先制 executeTurn(null, null, true); // 0.5秒後に敵の反撃（プレイヤーが倒してなければ） setTimeout(() => { if (enemy.hp > 0) executeTurn(null, null, false); }, 500); } else { // 敵が先制 executeTurn(null, null, false); // 0.5秒後にプレイヤーの反撃（自分が生きていれば） setTimeout(() => { if (curHP > 0) executeTurn(null, null, true); }, 500); } }, 1500); // ターンごとの間隔を少し広げて見やすく } function pAtk(e, log) { let dmg, msg, isCrit = Math.random() < myStats.crit; const useMagic = myStats.magicType || (Math.random() < (myStats.mp / 50)); if (useMagic) { dmg = Math.max(1, Math.floor(myStats.matk * (isCrit?1.5:1) - (e.mdef/2))); msg = `✨ ${playerName}の魔法！ ${dmg}ダメージ！`; } else { dmg = Math.max(1, Math.floor(myStats.atk * (isCrit?2:1) - (e.def/2))); msg = `⚔ ${playerName}の攻撃！ ${dmg}ダメージ！${isCrit?'★':''}`; } e.hp = Math.max(0, e.hp - dmg); document.getElementById('e-hp-bar').style.width = (e.hp / e.max * 100) + "%"; log.innerHTML += `
${msg}
`; return e.hp <= 0; } function eAtk(e, log) { // 回避判定 if (Math.random() < (myStats.spd - e.spd) * 0.05) { log.innerHTML += `
💨 ${playerName}はひらりと身をかわした！
`; return false; } let d = Math.max(1, Math.floor(e.atk - (myStats.def / 3))); curHP = Math.max(0, curHP - d); updateStatusDisplay(); // 左パネルを即更新 log.innerHTML += `
💥 敵の攻撃！ ${d}ダメージ！
`; return curHP <= 0; } function finish(intervalId, win) { clearInterval(intervalId); // インターバルを即座に破棄 setTimeout(() => { if (win) { alert("VICTORY! 150G獲得"); gold += 150; updateStatusDisplay(); showSub('menu-ui'); } else { // 敗北時はもっとシビアに alert("敗北しました…。所持金の一部を失い、村へ運ばれました。"); gold = Math.floor(gold * 0.7); // 30%失うペナルティ curHP = maxHP; updateStatusDisplay(); showSub('menu-ui'); } }, 600); }
